abstract class Utilizador {
  +Long id
  +Boolean ativo
  +DateTime criadoEm
  +DateTime atualizadoEm
}

class Tutor extends Utilizador {
  +String nome
  +String nif
  +String email
  +String contacto
  +String endereco
  +possuiAnimais() : Collection<Animal>
}

class Colaborador extends Utilizador {
  +String nome
  +String email
  +String contacto
  +PerfilAcesso perfilAcesso
  +temPermissao(operacao : String) : Boolean
}

class Animal {
  +Long id
  +String nome
  +Especie especie
  +String raca
  +Date dataNascimento
  +Double peso
  +String estadoSaude
  +String necessidadesAlimentares
  +String medicacaoEmCurso
  +String necessidadesEspeciais
}

class Alojamento {
  +Long id
  +TipoAlojamento tipo
  +Especie especieSuportada
  +EstadoLimpeza estadoLimpeza
  +String numero
  +Integer capacidade
  +Double precoDiario
  +estaDisponivel(dataInicio : Date, dataFim : Date) : Boolean
  +estaDisponivel() : Boolean
  +suportaEspecie(especie : Especie) : Boolean
  +podeSerReservado() : Boolean
}

class Reserva {
  +Long id
  +String codigo
  +Date dataInicio
  +Date dataFim
  +EstadoReserva estado
  +DateTime criadaEm
  +DateTime atualizadaEm
  +podeIniciarEstadia() : Boolean
  +estaCancelada() : Boolean
  +estaVencida() : Boolean
}

class Estadia {
  +Long id
  +String codigo
  +DateTime dataCheckIn
  +DateTime dataCheckOut
  +EstadoEstadia estado
  +DateTime criadaEm
  +DateTime encerradaEm
  +foiEncerrada() : Boolean
  +temPagamentoPendente() : Boolean
}

class Cuidado {
  +Long id
  +TipoCuidado tipo
  +String descricao
  +DateTime dataHora
  +DateTime criadoEm
  +pertenceAoMesmoTurno(outroCuidado : Cuidado) : Boolean
}

class Nota {

}

class ServicoExtra {
  +Long id
  +TipoServicoExtra tipo
  +String descricao
  +Double custo
  +DateTime data
  +DateTime criadoEm
  +podeSerAlterado() : Boolean
}

class Fatura {
  +Long id
  +String numero
  +DateTime dataEmissao
  +Double valorBase
  +Double desconto
  +Double total
  +EstadoFatura estado
  +String observacoes
  +DateTime criadaEm

  +temSaldoPendente() : Double
  +podeSerAlterada() : Boolean

class Pagamento {
  +Long id
  +Double valor
  +MetodoPagamento metodo
  +EstadoPagamento estado
  +Date dataVencimento
  +DateTime dataPagamento
  +String motivo
  +DateTime criadoEm
}

}

package "Services" {
  class ReservaService {
    +criarReserva(tutor, animal, alojamento, dataInicio, dataFim) : Reserva
    +confirmarReserva(reserva) : void
    +cancelarReserva(reserva) : void
    +consultarDisponibilidade(alojamento, dataInicio, dataFim, especie) : Boolean
  }

  class EstadiaService {
    +iniciarEstadia(reserva, colaborador) : Estadia
    +encerrarEstadia(estadia, colaborador) : void
  }

  class AlojamentoService {
    +verificarDisponibilidade(alojamento, dataInicio, dataFim) : Boolean
    +atualizarEstadoLimpeza(alojamento, novoEstado) : void
  }

  class FaturacaoService {
    +gerarFatura(estadia) : Fatura
    +registarPagamento(fatura, pagamento) : void
  }

  class AnimalService {
    +criarAnimal(tutor, dados) : Animal
    +consultarHistorialClinico(animal) : HistorialClinico
  }

  class UtilizadorService {
    +autenticar(username, password) : Utilizador
    +criarColaborador(dados, perfil) : Colaborador
  }
  
  class CuidadoService {}
  class RelatorioService {}
}

}
