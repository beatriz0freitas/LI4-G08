@startuml
title Diagrama de Sequência - Check-in
actor Rececionista
participant EstadiaService
participant ReservaRepository
participant EstadiaRepository
participant AlojamentoService

Rececionista -> EstadiaService: iniciarEstadia(reservaId, colaboradorId)
activate EstadiaService

EstadiaService -> ReservaRepository: findById(reservaId)
activate ReservaRepository
ReservaRepository --> EstadiaService: reserva
deactivate ReservaRepository

alt reserva.podeIniciarEstadia()
    EstadiaService -> EstadiaRepository: save(novaEstadia)
    activate EstadiaRepository
    EstadiaRepository --> EstadiaService: estadiaCriada
    deactivate EstadiaRepository

    EstadiaService -> AlojamentoService: atualizarEstado(alojamentoId, "Ocupado")
    activate AlojamentoService
    AlojamentoService --> EstadiaService
    deactivate AlojamentoService

    EstadiaService --> Rececionista: checkInRealizado
else
    EstadiaService --> Rececionista: Erro: Reserva inválida ou já utilizada
end
deactivate EstadiaService
@enduml
