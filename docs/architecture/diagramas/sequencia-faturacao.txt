@startuml
title Diagrama de Sequência - Faturação
actor Rececionista
participant FaturacaoService
participant FaturaRepository
participant PagamentoRepository

Rececionista -> FaturacaoService: registarPagamento(faturaId, dadosPagamento)
activate FaturacaoService

FaturacaoService -> FaturaRepository: findById(faturaId)
activate FaturaRepository
FaturaRepository --> FaturacaoService: fatura
deactivate FaturaRepository

alt fatura.temSaldoPendente() > 0
    FaturacaoService -> PagamentoRepository: save(novoPagamento)
    activate PagamentoRepository
    PagamentoRepository --> FaturacaoService: pagamentoRegistado
    deactivate PagamentoRepository

    FaturacaoService -> FaturaRepository: update(fatura.updateEstado())
    activate FaturaRepository
    FaturaRepository --> FaturacaoService
    deactivate FaturaRepository

    FaturacaoService --> Rececionista: Pagamento registado com sucesso
else
    FaturacaoService --> Rececionista: Erro: Fatura já está liquidada
end
deactivate FaturacaoService
@enduml
