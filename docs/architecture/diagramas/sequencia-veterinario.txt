@startuml
title Diagrama de Sequência - Intervenção do Veterinário
actor Veterinario
participant AnimalService
participant FaturacaoService
participant IntervencaoClinicaRepository
participant ServicoExtraRepository

Veterinario -> AnimalService: registarIntervencao(dadosIntervencao)
activate AnimalService

AnimalService -> IntervencaoClinicaRepository: save(novaIntervencao)
activate IntervencaoClinicaRepository
IntervencaoClinicaRepository --> AnimalService: intervencaoRegistada
deactivate IntervencaoClinicaRepository

alt dadosIntervencao.geraCusto
    AnimalService -> FaturacaoService: adicionarServicoExtra(estadiaId, dadosServico)
    activate FaturacaoService
    
    FaturacaoService -> ServicoExtraRepository: save(novoServicoExtra)
    activate ServicoExtraRepository
    ServicoExtraRepository --> FaturacaoService: servicoExtraRegistado
    deactivate ServicoExtraRepository

    FaturacaoService --> AnimalService
    deactivate FaturacaoService
end

AnimalService --> Veterinario: Intervenção registada com sucesso
deactivate AnimalService
@enduml
